namespace Foam.Bridges

theorem cantor_diagonal {S : Type} (s : S) (l : List S) :
    ∃ n, (playback l).at_ n ≠ (forever s).at_ n :=
  forever_escapes s l

theorem cantor_no_retraction {S : Type} (s : S) :
    ¬ ∃ g : CoList S → List S, ∀ c, playback (g c) = c :=
  playback_no_section s
